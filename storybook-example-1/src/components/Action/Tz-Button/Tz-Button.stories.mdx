import { Canvas, Meta, Story, ArgsTable } from '@storybook/addon-docs';
import { action } from '@storybook/addon-actions';
import { withDesign } from 'storybook-addon-designs';

import TzButton from './Tz-Button.vue';
import TzPayPalButton from './Tz-PayPalButton.vue';
import TzFilterIcon from '../../Informational/Icons/Tz-Filter-Icon.vue';

<Meta
  title="Components/Action/Button"
  component={TzButton}
  decorators={[withDesign]}
  argTypes={{
    type: {
      options: ['primary', 'secondary', 'finance'],
      control: { type: 'radio' }
    },
    size: {
      options: ['sm', 'md', 'lg'],
      control: { type: 'radio' }
    },
    iconPosition: {
      options: ['before', 'after'],
      control: { type: 'radio' }
    },
    label: {
      control: { type: 'text' },
      name: 'label'
    }
  }}
/>

export const Template = (args) => ({
  components: { TzButton },
  setup() {
    return { args };
  },
  template: '<tz-button v-bind="args" @click="clicked">{{ args.label }}</tz-button>',
  methods: {
    clicked: action('clicked')
  }
});

export const ButtonIconTemplate = (args) => ({
  components: { TzButton, TzCloseIcon },
  setup() {
    return { args };
  },
  template: `
    <tz-button v-bind="args">
        {{ args.label }} 
        <template #icon>
          <tz-close-icon class="w-5" />
        </template>
    </tz-button>
  `
});
