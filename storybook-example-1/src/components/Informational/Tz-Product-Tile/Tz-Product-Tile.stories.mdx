export const Template = (args) => ({
  components: { TzProductTile },
  setup: () => ({ args: reactive(args) }),
  template: `<tz-product-tile v-bind="args" @onFavourite="onFavourite" @onPin="onPin" @onRemove="onRemove" @onCompare="onCompare"></tz-product-tile>`,
  methods: {
    onPin(id, pinned) {
      this.args.pinned = !this.args.pinned;
      action('onPin')(id, pinned);
    },
    onFavourite(id, favourite) {
      this.args.favourite = !this.args.favourite;
      action('onFavourite')(id, favourite);
    },
    onRemove: action('onRemove'),
    onCompare: action('onCompare')
  }
});

# Product Tile

The product tiles are a key sales componet that allow the user to browse many products at once before commiting to a product page, these tiles are dynamically managed by the CMS, they have vairous types and differ in functionality subject to type.

Bikes for example have the most complex functionality and tools associated with the ability to compare.
